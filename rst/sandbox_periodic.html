def _getProp(X, E1, E2, nu1, nu2):
    X_prop = np.zeros(X.shape + (2,))
    X_prop[...,0] = np.where(X == 1, E1, E2)
    X_prop[...,1] = np.where(X == 1, nu1, nu2)
    return X_prop

def getDel(L):
    C = (L-1)/2
    Del = np.zeros((2, L, L))
    Del[0, C, C] = 1
    Del[1] = np.abs(Del[0]-1)
    return Del
